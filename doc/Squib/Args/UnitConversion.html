def parse(arg, dpi=300)
  case arg.to_s.rstrip
  when /in$/ #ends with "in"
    arg.rstrip[0..-2].to_f * dpi
  when /cm$/ #ends with "cm"
    arg.rstrip[0..-2].to_f * dpi * INCHES_IN_CM
  else
    arg
  end
end
